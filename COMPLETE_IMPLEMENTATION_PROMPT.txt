UPLOAD WORKFLOW:
1. User drops file or clicks to browse
2. System validates CSV/JSON structure
3. Preview table shows first 5 rows with validation status
4. User can edit individual rows in preview
5. User clicks "Upload" to process all valid rows
6. System shows progress bar during upload
7. Success/error summary displayed

import { useDropzone } from 'react-dropzone';

export function BulkUploadQuestionsDialog({ componentId, onSuccess }) {
  const [open, setOpen] = useState(false);
  const [format, setFormat] = useState<'csv' | 'json'>('csv');
  const [file, setFile] = useState<File | null>(null);
  const [preview, setPreview] = useState<any[]>([]);
  const [uploading, setUploading] = useState(false);
  
  const { getRootProps, getInputProps } = useDropzone({

    onDrop: async (acceptedFiles) => {

      setFile(file);

          body: JSON.stringify({
            format,
            data: format === 'csv' ? text : JSON.parse(text),
            validateOnly: true
          })
        }
      );
      
      const result = await response.json();
      setPreview(result);
    }
  });
  
  const handleUpload = async () => {
    if (!file) return;
    setUploading(true);
    
    try {
      const text = await file.text();
      
      const response = await fetch(
        `/api/assessments/admin/components/${componentId}/questions/bulk-upload`,
        {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({

            data: format === 'csv' ? text : JSON.parse(text),
            validateOnly: false
          })
        }

      const result = await response.json();
      
      if (result.success) {

      setUploading(false);

    a.href = url;
    a.download = 'questions_template.csv';
    a.click();

  return (
    <Dialog open={open} onOpenChange={setOpen}>
      <Button variant="outline" onClick={() => setOpen(true)}>
        <Upload className="mr-2 h-4 w-4" />
        Bulk Upload Questions
      </Button>
      
      <DialogContent className="max-w-3xl">
        <DialogHeader>

          <TabsContent value="csv" className="space-y-4">
            <div className="flex justify-end">
              <Button variant="ghost" size="sm" onClick={downloadTemplate}>
                <Download className="mr-2 h-4 w-4" />

              </Button>
            </div>
            
            <div
              {...getRootProps()}
              className="border-2 border-dashed rounded-lg p-8 text-center cursor-pointer"
            >
              <input {...getInputProps()} />
              <FileText className="mx-auto h-12 w-12 text-muted-foreground" />

              {file && <p className="mt-2 text-sm">{file.name}</p>}

BACKEND API IMPLEMENTATION:
```typescript
// app/api/assessments/admin/components/[componentId]/questions/bulk-upload/route.ts
import { NextRequest, NextResponse } from 'next/server';
import { parse } from 'csv-parse/sync';
import { prisma } from '@/lib/db';
import { z } from 'zod';

const questionSchema = z.object({
  question_text: z.string().min(10),
  question_type: z.enum(['MULTIPLE_CHOICE', 'TRUE_FALSE', 'CODING_CHALLENGE', 'ESSAY', 'SCENARIO_BASED']),
  points: z.number().optional().default(1),
  time_limit: z.number().optional(),
});

export async function POST(
  request: NextRequest,
  { params }: { params: { componentId: string } }
) {
  try {
    const body = await request.json();
    const { format, data, validateOnly } = body;
    
    let questions = [];
    
    if (format === 'csv') {
      questions = parse(data, {
        columns: true,
        skip_empty_lines: true,
        cast: true
      });
    } else {
      questions = data.questions;
    }
    
    // Validate each question
    const validationResults = questions.map((q, idx) => {
      try {
        questionSchema.parse(q);
        return { valid: true, row: idx + 1 };
      } catch (error) {
        return { valid: false, row: idx + 1, errors: error.errors };
      }
    });
    
    const validQuestions = validationResults.filter(v => v.valid);
    const invalidQuestions = validationResults.filter(v => !v.valid);
    
    if (validateOnly) {
      return NextResponse.json({
        valid: invalidQuestions.length === 0,
        validCount: validQuestions.length,
        invalidCount: invalidQuestions.length,
        errors: invalidQuestions
      });
    }
    
    // Insert valid questions
    const createdQuestions = await prisma.$transaction(
      validQuestions.map((_, idx) => 
        prisma.componentQuestion.create({
          data: {
            componentId: params.componentId,
            questionText: questions[idx].question_text,
            questionType: questions[idx].question_type,
            points: questions[idx].points || 1,
            timeLimit: questions[idx].time_limit,
            order: questions[idx].order || idx + 1,
            options: questions[idx].options,
          }
        })
      )
    );
    
    return NextResponse.json({
      success: true,
      summary: {
        totalRows: questions.length,
        validRows: validQuestions.length,
        invalidRows: invalidQuestions.length,
        successfulUploads: createdQuestions.length,
        failedUploads: 0
      },
      errors: invalidQuestions,
      createdQuestions
    });
    
  } catch (error) {
    return NextResponse.json(
      { error: 'Bulk upload failed', details: error.message },
      { status: 500 }
    );
  }
}
```

1.3 CONFIGURE RUNTIME GENERATION
--------------------------------------------------------------------------------

BUTTON SPECIFICATION:
- Label: "Configure Runtime Generation"
- Icon: Zap icon (dynamic/runtime indicator)
- Position: Third button in row
- Style: Outline variant with accent color

CONCEPT:
Generate questions dynamically DURING assessment execution, powered by AI Agent
that adapts based on respondent's performance.

KEY DIFFERENCE:
- Feature 1.2: Pre-generates questions for question bank
- Feature 1.3: Generates questions on-the-fly during assessment

CONFIGURATION FORM:

SECTION 1: Generation Triggers
- Enable Runtime Generation (toggle)
  
- Generation Mode (radio buttons)
  ○ Adaptive (generates based on performance)
  ○ Progressive (increases difficulty as user advances)
  ○ Gap-filling (targets weak areas)
  ○ Exploratory (probes understanding depth)
  
- Trigger Conditions (multi-select)
  □ User answers correctly 3 times → increase difficulty
  □ User answers incorrectly 2 times → provide easier question
  □ Time spent < 50% of limit → question too easy
  □ Competency score < 60% → generate more for that competency

SECTION 2: Generation Constraints
- Max Questions to Generate (1-20, default: 10)
- Question Pool Size (Small/Medium/Large/Unlimited)
- Response Time Limit (seconds, default: 30)
- Fallback Strategy:
  ○ Skip question if generation fails
  ○ Show pre-defined fallback question
  ○ End assessment gracefully
  ○ Retry once then skip

SECTION 3: Context for Generation
Assessment Context (checkboxes - what AI receives):
  ☑ Component/topic being assessed
  ☑ User's current score
  ☑ Questions already answered
  ☑ Time spent on previous questions
  ☑ Difficulty level of previous questions
  ☑ User's profile (role, experience level)
  ☑ Learning objectives

Competency Mapping (table):
| Competency          | Weight | Min Q | Max Q |
|---------------------|--------|-------|-------|
| Python Programming  | 40%    | 2     | 5     |
| Problem Solving     | 30%    | 1     | 3     |
| Code Optimization   | 30%    | 1     | 3     |

Performance Thresholds:
- Score threshold for increasing difficulty: 80%
- Score threshold for decreasing difficulty: 50%
- Minimum questions before adaptation: 3

SECTION 4: Question Generation Prompts
Base Prompt Template (textarea):
```
You are assessing a candidate on {topic}.

Current Context:
- Difficulty Level: {currentDifficulty}
- Questions Answered: {questionsAnswered}
- Current Score: {currentScore}%
- Weak Areas: {weakAreas}

Generate a {questionType} question that:
- Tests {specificCompetency}
- Is {difficultyAdjustment} than previous questions
- Takes approximately {estimatedTime} seconds
```

SECTION 5: Quality Assurance
Auto-validation (checkboxes):
  ☑ Check for duplicate questions
  ☑ Validate question clarity (using AI)
  ☑ Ensure answer is unambiguous
  ☑ Verify difficulty aligns with intent

Human Review Threshold:
  ○ None (full automation)
  ○ Random 10% sample
  ○ All questions (for testing)
  ○ Only when AI confidence < 80%

SECTION 6: LangGraph Agent Configuration
- Agent Model: GPT-4 Turbo / GPT-3.5 / Claude 3
- Agent Temperature: 0.7 (slider 0-1)
- Agent Max Tokens: 1000 (range: 500-2000)
- Use LangGraph StateGraph: ON (always)
- Redis State Key Pattern: 
  `runtime:component:{componentId}:session:{sessionId}`

RUNTIME GENERATION FLOW:

1. Adaptive Flow Agent determines next question needed
2. Checks if RuntimeGenerationConfig enabled for component
3. Calls Runtime Question Generation Agent
4. Agent constructs prompt from template + current context
5. Calls LLM to generate question
6. Validates generated question
7. Stores temporarily (not in permanent bank)
8. Updates state in Redis
9. Returns question to assessment interface
10. After assessment, admin can review and approve good questions

export async function POST(request: NextRequest) {
  try {
    const data = await request.json();
    
    // Check if company email exists
    const existingClient = await prisma.client.findUnique({
      where: { email: data.companyEmail }
    });
    
    if (existingClient) {
      return NextResponse.json(
        { error: 'Company email already registered' },
        { status: 400 }
      );
    }
    
    // Check if admin email exists
    const existingUser = await prisma.user.findUnique({
      where: { email: data.adminEmail }
    });
    
    if (existingUser) {
      return NextResponse.json(
        { error: 'Admin email already registered' },
        { status: 400 }
      );
    }
    
    const hashedPassword = await bcrypt.hash(data.adminPassword, 12);
    
    // Create client and admin user in transaction
    const result = await prisma.$transaction(async (tx) => {
      const client = await tx.client.create({
        data: {
          name: data.companyName,
          slug: generateSlug(data.companyName),
          email: data.companyEmail,
          phone: data.companyPhone,
          website: data.companyWebsite,
          industry: data.industry,
          companySize: data.companySize,
          country: data.country,
          address: data.address,
          plan: data.plan,
          isActive: false, // Activate after email verification
          subscriptionStart: new Date(),
          maxProjects: data.plan === 'STARTER' ? 5 : 
                       data.plan === 'PROFESSIONAL' ? 20 : 999,
          maxUsers: data.plan === 'STARTER' ? 50 : 
                    data.plan === 'PROFESSIONAL' ? 200 : 9999,
          createdBy: 'SELF_REGISTRATION'
        }
      });
      
      const adminUser = await tx.user.create({
        data: {
          email: data.adminEmail,
          password: hashedPassword,
          name: `${data.adminFirstName} ${data.adminLastName}`,
          role: 'CLIENT_ADMIN',
          accountType: 'CLIENT_ADMIN',
          clientId: client.id,
          isActive: false,
          emailVerified: null
        }
      });
      
      await tx.clientSettings.create({
        data: {
          clientId: client.id,
          primaryColor: '#3b82f6',
          secondaryColor: '#10b981',
          enableDepartments: true,
          enableManagers: true,
          enableCustomTemplates: data.plan !== 'STARTER'
        }
      });
      
      return { client, adminUser };
    });
    
    // Send verification email
    await sendVerificationEmail(data.adminEmail, result.adminUser.id);
    
    return NextResponse.json({
      success: true,
      clientId: result.client.id,
      message: 'Registration successful. Check email for verification.'
    });
    
  } catch (error) {
    return NextResponse.json(
      { error: 'Registration failed', details: error.message },
      { status: 500 }
    );
  }
}

function generateSlug(name: string): string {
  return name
    .toLowerCase()
    .replace(/[^a-z0-9]+/g, '-')
    .replace(/^-+|-+$/g, '');
}

```typescript
export async function POST(
  request: NextRequest,
  { params }: { params: { clientSlug: string } }
) {
  try {
    const { csvData } = await request.json();
    
    const client = await prisma.client.findUnique({
      where: { slug: params.clientSlug }
    });
    
    if (!client) {
      return NextResponse.json({ error: 'Client not found' }, { status: 404 });
    }
    
    const rows = parse(csvData, {
      columns: true,
      skip_empty_lines: true
    });
    
    const validRows = [];
    const errors = [];
    
    for (const [index, row] of rows.entries()) {
      const validation = validateEmployeeRow(row);
      if (validation.valid) {
        validRows.push(row);
      } else {
        errors.push({ row: index + 2, message: validation.error });
      }
    }
    
    const createdEmployees = await prisma.$transaction(
      validRows.map(row => {
        const tempPassword = generateTempPassword();
        const hashedPassword = bcrypt.hashSync(tempPassword, 12);
        
        return prisma.user.create({
          data: {
            email: row.email,
            password: hashedPassword,
            name: `${row.first_name} ${row.last_name}`,
            role: 'ASSESSOR',
            accountType: 'CLIENT_USER',
            clientId: client.id,
            employeeId: row.employee_id,
            department: row.department,
            isActive: true
          }
        }).then(async (user) => {
          await prisma.assessorProfile.create({
            data: {
              userId: user.id,
              employeeId: row.employee_id,
              departmentUnit: row.department,
              currentDesignation: row.designation,
              reportingManager: row.reporting_manager_email || '',
              dateOfJoining: new Date(row.date_of_joining),
              yearsInCurrentRole: 0,
              educationalQual: row.educational_qualification || '',
              professionalCerts: row.professional_certifications?.split('|') || []
            }
          });
          
          await sendWelcomeEmail({
            email: user.email,
            name: user.name,
            tempPassword,
            clientName: client.name,
            loginUrl: `${process.env.NEXT_PUBLIC_BASE_URL}/clients/${client.slug}/login`
          });
          
          return user;
        });
      })
    );
    
    return NextResponse.json({
      success: true,
      created: createdEmployees.length,
      errors: errors.length > 0 ? errors : undefined
    });
    
  } catch (error) {
    return NextResponse.json(
      { error: 'Upload failed', details: error.message },
      { status: 500 }
    );
  }
}

function generateTempPassword(): string {
  const chars = 'ABCDEFGHJKLMNPQRSTUVWXYZabcdefghijkmnpqrstuvwxyz23456789!@#$%^&*';
  let password = '';
  for (let i = 0; i < 12; i++) {
    password += chars.charAt(Math.floor(Math.random() * chars.length));
  }
  return password;
}

function calculateGap(currentLevel: number, requiredLevel: number) {
  const gap = requiredLevel - currentLevel;
  
  let level: 'HIGH' | 'MEDIUM' | 'LOW' | 'NONE';
  let color: string;
  
  if (gap >= 3) {
    level = 'HIGH';
    color = 'red';
  } else if (gap >= 2) {
    level = 'MEDIUM';
    color = 'yellow';
  } else if (gap >= 1) {
    level = 'LOW';
    color = 'green';
  } else {
    level = 'NONE';
    color = 'gray';
  }
  
  return { gap, level, color };
}
```

import { OpenAI } from 'openai';

export async function POST(request: NextRequest) {
  try {
    const { competencyName, category, targetLevel, description, context } = 
      await request.json();
    
    const prompt = `Generate competency definition with behavioral indicators...`;
    
    const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
    
    const completion = await openai.chat.completions.create({
      model: "gpt-4-turbo-preview",
      messages: [
        {
          role: "system",
          content: "Expert in competency modeling and behavioral assessment."
        },
        {
          role: "user",
          content: prompt
        }
      ],
      response_format: { type: "json_object" },
      temperature: 0.7
    });
    
    const generated = JSON.parse(completion.choices[0].message.content);
    
    return NextResponse.json(generated);
    
  } catch (error) {
    return NextResponse.json(
      { error: 'Generation failed' },
      { status: 500 }
    );
  }
}

WORKFLOW:
```
Step 1: Analyze Role
- Get all role competencies
- Get behavioral indicators for each
- Determine required proficiency levels

Step 2: Generate Components (for each competency)
- Create AssessmentComponent
- Component name: "{Competency Name} Assessment"
- Duration: Based on indicator count
- Difficulty: Based on required level

Step 3: Generate Questions (for each component)
- Based on positive indicators → "What would you do?" scenarios
- Based on negative indicators → "What would you avoid?" questions
- Mix of MCQ, scenario-based, essay questions
- 5-10 questions per component

Step 4: Create Assessment Model
- Model name: "{Role Name} Competency Assessment"
- Bundle all components
- Set weightage based on competency criticality
- Total duration: Sum of component durations

Step 5: Link Everything
- Create CompetencyComponentMapping records
- Link role to assessment model
- Mark as "Role-Based Assessment"
```
